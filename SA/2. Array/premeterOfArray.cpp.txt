// Print Parameter of this array.
#include <iostream>
#include<vector>
using namespace std;

int main() {
    // Write C++ code here
    vector<vector<int>> A {
        {1, 2, 3, 10},
        {4, 5, 6, 12},
        {7, 8, 9, 13},
        {10, 11, 12, 18}
    };
    
    int N = A.size();
    vector<int> result((N-1)*4, 0);
    int i=0, j=0, k=0;
    // First row
    for(int l=0; l<N-1; l++) { // 3
        result[k] = A[i][j];
        k++; j++;
    }
    
    // Last Column i = 0, j = 3
    for(int l=0; l<N-1; l++) { // 3
        result[k] = A[i][j];
        k++; i++;
    }
    
    // last row i = 3 and J = 3
    for(int l=0; l<N-1; l++) { // 3
        result[k] = A[i][j];
        k++; j--;
    }
    
    // last row i = 3 and J = 0
    for(int l=0; l<N-1; l++) { // 3
        result[k] = A[i][j];
        k++; i--;
    }
    
    for(int i=0; i<result.size(); i++) {
        cout<<result[i]<<" ";
    }

    return 0;
}
